# intiialise image size
size(1920,1080)

$x = 0
$y = 540

while $x < 1920{

    # if x is even
    if ($x % 2 == 0){
        $y = $y+$x/2
    }else{
        $y = $y-$x/2
    }

    pos($x, $y)
    $c = $x*255/1920
    fill(255,$c,255-$c)

    ellipse(10)

    $x = $x + 19
}

fill("none")
stroke("black")
line()
