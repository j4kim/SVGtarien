# intiialise image size
$w = 1920
$h = 1080
size($w, $h)

# draw axis
stroke("blue")
pos($w/2,0)
move(0,$h)
line(2) # draw a line with the 2 last points
pos(0,$h/2)
move($w,0)
line(2)
clean() # clean positions history

$x = -2
$y = 540

while $x < 2{

    $y = 540+ ($x ^ 3)*100

    pos(($x+2)*($w/4), $y)
    $c = $x*255/1920
    fill(255,$c,255-$c)

    ellipse(10)

    $x = $x + 0.1
}

fill("none")
stroke("black")
line()
